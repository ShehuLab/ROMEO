142  virtual bool IsAcceptable(Cfg & cfg)
143  {
144  if(cfg.IsEnergySet() == false)
145  cfg.SetEnergy(GetCfgEnergyEvaluator()->EvaluateEnergy(cfg));
146 
147 
148  if (cfg.GetEnergy() > m_maxEnergy)
149  m_maxEnergy = cfg.GetEnergy();
150 
151  if (cfg.GetEnergy() < m_minEnergy)
152  m_minEnergy = cfg.GetEnergy();
153 
154  // Metropolis-test
155  bool passMMC = true;
156 
157  double deltaE = cfg.GetEnergy() - m_cfgSource->GetEnergy();
158  double randomDraw = 0.0;
159  double acceptProb = 1.0;
160 
161  if (deltaE > 0.0)
162  {
163  acceptProb = exp(-1 * deltaE/(Constants::VAL_Boltzman_K * m_temperature));
164 
165  randomDraw = RandomUniformReal();
166  if (acceptProb < randomDraw)
167  {
168  passMMC = false;
169  }
170  }
171 
172  (passMMC) ? ++m_testSuccessCount : ++m_testFailCount;
173  if (m_verboseFlag)
174  {
175  Logger::m_out << "MMCTest passMMC:" << passMMC
176  << " temp:" << m_temperature
177  << " deltaE:" << deltaE
178  << " new cfg energy:" << cfg.GetEnergy()
179  << " prob:" << acceptProb
180  << " random:" << randomDraw << "\n";
181  }
182  return (passMMC);
183  }
