27  double usrSimilarity(const double* p1, const double* p2, const int size)
28  {
29  assert(p1 && p2 && size > 0);
30 
31  double d = 0.0;
32  for(int i = 0; i < size; ++i)
33  d += fabs(p2[i] - p1[i]); //Manhattan distance
34 
35  return 1.0/(1.0 + 1.0/size * d);
36  }
