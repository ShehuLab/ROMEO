157  virtual bool IsAcceptable(Cfg & cfg)
158  {
159  if(cfg.IsEnergySet() == false)
160  cfg.SetEnergy(GetCfgEnergyEvaluator()->EvaluateEnergy(cfg));
161 
162 
163  if (cfg.GetEnergy() > m_maxEnergy)
164  m_maxEnergy = cfg.GetEnergy();
165 
166  if (cfg.GetEnergy() < m_minEnergy)
167  m_minEnergy = cfg.GetEnergy();
168 
169  // Metropolis-test
170  bool passMMC = true;
171 
172  double deltaE = cfg.GetEnergy() - m_cfgSource->GetEnergy();
173  double priorTemp = m_temperature;
174  double randomDraw = 0.0;
175  double acceptProb = 1.0;
176 
177 
178 
179  if (deltaE > 0.0)
180  {
181  acceptProb = exp(-1 * deltaE/(Constants::VAL_Boltzman_K * m_temperature));
182 
183  randomDraw = RandomUniformReal();
184  if (acceptProb < randomDraw)
185  {
186  passMMC = false;
187  /* adjust temperature up to increase change of success */
188  m_temperature *= m_temperatureAdjRate;
189 
190  }
191  else
192  {
193  /* SUCCESS, so, decrease the temperature */
194  // m_temperature /= pow(2, (deltaE) / (0.1 * m_maxEnergy - m_minEnergy));
195  m_temperature /= m_temperatureAdjRate;
196  }
197  }
198  else
199  {
200  m_temperature /= m_temperatureAdjRate;
201  }
202 
203  (passMMC) ? ++m_testSuccessCount : ++m_testFailCount;
204 
205  Logger::m_out << "MMCTest passMMC:" << passMMC
206  << ":old_temp:" << priorTemp
207  << ":new_temp:" << m_temperature
208  << ":deltaE:" << deltaE
209  << ":new cfg energy:" << cfg.GetEnergy()
210  << ":prob:" << acceptProb
211  << ":random:" << randomDraw
212  << ":adjRate:" << m_temperatureAdjRate << std::endl;
213  return (passMMC);
214  }
